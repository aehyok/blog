nginx 配置三个ocelot节点 ​

  • 这三个ocelot节点可以加入到consul中，并添加健康检查的接口，供consul进行服务注册和健康检查
server {
  listen 8500;
  location / {
    proxy_pass http://xdpconsul;
    proxy_redirect default;
  }
}

upstream xdpconsul {
  server 116.204.98.9:5003;
  server 116.204.98.9:5030;
  server 116.204.98.9:5300;
}
